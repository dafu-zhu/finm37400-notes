# FINM 37400 Notes Website — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Stand up a public Quarto website at `https://dafu-zhu.github.io/finm37400-notes/` that hosts the converted FINM 37400 lecture notes (8 lectures, 56 figures, custom math macros, callouts, and an interactive chapter dependency map) with auto-deploy from `main` via GitHub Actions.

**Architecture:** Source `.tex` (snapshotted from `D:/GitHub/good-student-note/notes/finm37400`) is converted to Quarto `.qmd` by a Python script (`scripts/convert.py`) using a pre-pass → pandoc → post-pass pipeline. TikZ diagrams are pre-rendered to PNGs via matplotlib (`scripts/render_tikz.py`). Quarto renders to static HTML with KaTeX math; GH Actions builds and deploys to GitHub Pages.

**Tech Stack:** Quarto 1.5.57, pandoc 3.9 (bundled in Quarto), Python 3.11 + matplotlib + numpy, KaTeX, GitHub Actions, gh CLI.

**Reference spec:** `docs/superpowers/specs/2026-05-05-finm37400-website-design.md`

```python
"""Lec 8 TikZ #1 — implied volatility smile (3 curves + ATM dots).

Reproduces the diagram at source-tex/lec08.tex L220-L234.
"""
from pathlib import Path
import matplotlib.pyplot as plt
import numpy as np

OUT = Path(__file__).resolve().parents[2] / "figures" / "L8_tikz_1.png"

def render():
    fig, ax = plt.subplots(figsize=(7, 4), dpi=200)
    x = np.linspace(0.3, 7.7, 200)
    blue  = 1.7 + 0.10 * (x - 3.8) ** 2
    red   = 2.4 + 0.10 * (x - 2.8) ** 2
    green = 1.2 + 0.10 * (x - 4.8) ** 2

    ax.plot(x, blue,  color="tab:blue",  linewidth=2)
    ax.plot(x, red,   color="tab:red",   linewidth=2)
    ax.plot(x, green, color="tab:green", linewidth=2)

    pts = [(3.8, 1.70, "tab:blue",  "ATM today",          "below"),
           (2.8, 2.40, "tab:red",   r"ATM if forward $\downarrow$", "above"),
           (4.8, 1.20, "tab:green", r"ATM if forward $\uparrow$",   "below")]
    for xv, yv, c, label, where in pts:
        ax.scatter([xv], [yv], color=c, s=40, zorder=3)
        offset = -0.3 if where == "below" else 0.3
        va = "top" if where == "below" else "bottom"
        ax.annotate(label, (xv, yv), xytext=(xv, yv + offset),
                    color=c, ha="center", va=va, fontsize=9)

    ax.set_xlabel("strike $K$")
    ax.set_ylabel(r"implied vol $\sigma$")
    ax.set_xlim(0, 8.5)
    ax.set_ylim(0, 4.6)
    ax.spines["top"].set_visible(False)
    ax.spines["right"].set_visible(False)
    fig.tight_layout()
    OUT.parent.mkdir(parents=True, exist_ok=True)
    fig.savefig(OUT, dpi=200, bbox_inches="tight")
    plt.close(fig)
    print(f"  wrote {OUT.relative_to(OUT.parents[2])}")

if __name__ == "__main__":
    render()
```

```python
"""Lec 8 TikZ #2 — 1-period stock binomial tree."""
from pathlib import Path
import matplotlib.pyplot as plt

OUT = Path(__file__).resolve().parents[2] / "figures" / "L8_tikz_2.png"

def _node(ax, x, y, text, color="black"):
    ax.annotate(
        text, (x, y), ha="center", va="center", color=color, fontsize=12,
        bbox=dict(boxstyle="round,pad=0.4", fc="white", ec=color, lw=1.2),
    )

def _arrow(ax, x0, y0, x1, y1, label, where):
    ax.annotate("", xy=(x1, y1), xytext=(x0, y0),
                arrowprops=dict(arrowstyle="->", lw=1.5, color="black"))
    mx, my = (x0 + x1) / 2, (y0 + y1) / 2
    dy = 0.18 if where == "above" else -0.18
    va = "bottom" if where == "above" else "top"
    ax.text(mx, my + dy, label, ha="center", va=va, fontsize=10)

def render():
    fig, ax = plt.subplots(figsize=(6, 3.5), dpi=200)
    _node(ax, 0, 0, "$S_0 = 100$")
    _node(ax, 3, 1.2, "$S_u = 110$", color="tab:blue")
    _node(ax, 3, -1.2, "$S_d = 90$", color="tab:red")
```

```python
"""Lec 8 TikZ #4 — multiperiod CRR recombining tree (depth 2)."""
from pathlib import Path
import matplotlib.pyplot as plt

from lec08_02 import _node

OUT = Path(__file__).resolve().parents[2] / "figures" / "L8_tikz_4.png"

NODES = [
    (0,  0,    "$S_0$",          "black"),
    (2,  1.4,  "$S_u$",          "black"),
    (2, -1.4,  "$S_d$",          "black"),
    (4,  2.8,  "$S_{uu}$",       "black"),
    (4,  0,    "$S_{ud}=S_0$",   "black"),
    (4, -2.8,  "$S_{dd}$",       "black"),
]
EDGES = [(0, 1), (0, 2), (1, 3), (1, 4), (2, 4), (2, 5)]
PAYOFFS = [
    (5.7,  2.8, r"$f_{uu}=\max(S_{uu}-K,0)$"),
    (5.7,  0,   r"$f_{ud}=\max(S_{ud}-K,0)$"),
    (5.7, -2.8, r"$f_{dd}=\max(S_{dd}-K,0)$"),
]

def render():
    fig, ax = plt.subplots(figsize=(8, 5), dpi=200)
    for x, y, label, color in NODES:
        _node(ax, x, y, label, color=color)
    for i, j in EDGES:
        x0, y0, *_ = NODES[i]
        x1, y1, *_ = NODES[j]
        ax.annotate("", xy=(x1 - 0.18, y1), xytext=(x0 + 0.18, y0),
                    arrowprops=dict(arrowstyle="->", lw=1.3, color="black"))
    for x, y, txt in PAYOFFS:
        ax.text(x, y, txt, color="tab:blue", fontsize=9, va="center")
    ax.text(-0.6, -3.6, "time grid:", fontsize=9)
    for x, label in [(0, "$0$"), (2, r"$\Delta t$"), (4, r"$2\Delta t$")]:
        ax.text(x, -3.6, label, ha="center", fontsize=9)
    ax.set_xlim(-1.0, 9.5)
```

### Task 22: Write `scripts/convert.py`

- [ ] **Step 1: Write file**

```python
"""Convert source-tex/*.tex to lectures/*.qmd.

Pipeline per file:
  1. Pre-pass:  drop checkpoints, swap color blocks for callouts, replace
                tikz with image refs, strip per-lecture LaTeX bookkeeping,
                extract part title.
  2. Pandoc:    LaTeX -> Markdown with --katex math passthrough.
  3. Post-pass: minor cleanup of pandoc artifacts.
  4. YAML wrap.

Idempotent: overwrites lectures/lec0N.qmd on every run.
"""
from __future__ import annotations
import re
import subprocess
import sys
from pathlib import Path

ROOT = Path(__file__).resolve().parents[1]
SRC = ROOT / "source-tex"
OUT = ROOT / "lectures"

BLOCK_MAP = {
    "keyblock":      ("important", "Key concept"),
    "exerciseblock": ("caution",   "Exercise"),
    "noteblock":     ("note",      "Note"),
    "tipblock":      ("tip",       "Filling the gap"),
}

CHECKPOINT_RE = re.compile(
    r"\\begin\{checkpoint\}.*?\\end\{checkpoint\}", re.DOTALL
)
TIKZ_RE = re.compile(
    r"\\begin\{tikzpicture\}.*?\\end\{tikzpicture\}", re.DOTALL
)
PART_RE = re.compile(r"\\part\*\{Lecture\s+(\d+):\s*([^}]+)\}")


def remove_checkpoint_blocks(tex: str) -> str:
    return CHECKPOINT_RE.sub("", tex)


def convert_color_blocks(tex: str) -> str:
    for env, (callout, title) in BLOCK_MAP.items():
        tex = re.sub(
            rf"\\begin\{{{env}\}}",
            f':::{{.callout-{callout} title="{title}"}}\n',
            tex,
        )
        tex = re.sub(rf"\\end\{{{env}\}}", "\n:::", tex)
    return tex


def replace_tikz_blocks(tex: str, lecture_num: int) -> tuple[str, int]:
    counter = {"i": 0}

    def sub(_m):
        counter["i"] += 1
        return f"![](../figures/L{lecture_num}_tikz_{counter['i']}.png)"

    return TIKZ_RE.sub(sub, tex), counter["i"]


def strip_lecture_macros(tex: str) -> str:
    tex = re.sub(r"\\beginlecture\{\d+\}", "", tex)
    tex = re.sub(r"\\addcontentsline\{toc\}\{part\}\{[^}]*\}", "", tex)
    tex = tex.replace("\\textperiodcentered{}", "·")
    return tex


def extract_part_title(tex: str) -> tuple[int, str]:
    m = PART_RE.search(tex)
    if not m:
        raise ValueError("no \\part*{Lecture N: ...} found")
    return int(m.group(1)), m.group(2).strip()


def strip_part(tex: str) -> str:
    return PART_RE.sub("", tex)


def run_pandoc(tex: str) -> str:
    proc = subprocess.run(
        ["pandoc", "-f", "latex", "-t", "markdown",
         "--wrap=preserve", "--katex"],
        input=tex,
        capture_output=True,
        text=True,
        encoding="utf-8",
        check=True,
    )
    return proc.stdout


def clean_pandoc_output(md: str) -> str:
    # Pandoc escapes underscores in image paths under markdown output;
    # reverse that for the figure refs we generated.
    md = re.sub(
        r"!\[\]\((\.\./figures/[^)]+)\)",
        lambda m: "![](" + m.group(1).replace("\\_", "_") + ")",
        md,
    )
    return md


def wrap_with_yaml(md: str, title: str, lecture_num: int) -> str:
    header = (
        "---\n"
        f'title: "Lecture {lecture_num}: {title}"\n'
        "number-sections: true\n"
        "---\n\n"
        "<!-- AUTO-GENERATED by scripts/convert.py — do not edit. -->\n\n"
    )
    return header + md


def convert_file(tex_path: Path) -> None:
    raw = tex_path.read_text(encoding="utf-8")
    lecture_num, title = extract_part_title(raw)
    tex = strip_part(raw)
    tex = remove_checkpoint_blocks(tex)
    tex = convert_color_blocks(tex)
    tex, tikz_n = replace_tikz_blocks(tex, lecture_num)
    tex = strip_lecture_macros(tex)
    md = run_pandoc(tex)
    md = clean_pandoc_output(md)
    md = wrap_with_yaml(md, title, lecture_num)
    out = OUT / f"lec{lecture_num:02d}.qmd"
    out.write_text(md, encoding="utf-8")
    print(f"  {tex_path.name} -> {out.name} ({tikz_n} tikz)")


def main():
    OUT.mkdir(parents=True, exist_ok=True)
    sources = sorted(SRC.glob("lec*.tex"))
    if not sources:
        print("no source .tex files in source-tex/", file=sys.stderr)
        sys.exit(1)
    for tex in sources:
        convert_file(tex)
    print(f"done: {len(sources)} lectures")


if __name__ == "__main__":
    main()
```
